// Hacker
string hack(string &n){
   vector<int>v(10,0);
   for(int i=0;i<n.size();i++){
      v[n[i]-'0']++;
   }
   string ans="";
   for(int i=9;i>=0;i--){
      while(v[i]--){
         ans+=to_string(i);
      }
   }
   return ans;
}
