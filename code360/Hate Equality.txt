// Hate Equality
#include <bits/stdc++.h>
int canYouMakeDifference(int n, string &s) {
    vector<int>v(26,0);
    int all=n;
    for(int i=0;i<n;i++){
        v[s[i]-'a']++;
    }
    for(int i=0;i<n;i++){
        bool ok=0;
        if(all-v[s[i]-'a']==0){
            return 0;
        }
    }
    return 1;
}
