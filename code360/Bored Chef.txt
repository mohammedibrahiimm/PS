// Bored Chef
int boredChef(int n, int k, string &s) {
   vector<int>v(26,0);
   for(int i=0;i<n;i++){
      v[s[i]-'a']++;
      if(v[s[i]-'a']>=k){
         return 1;
      }
   }
   return 0;
}
