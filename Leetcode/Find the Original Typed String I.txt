// 3330. Find the Original Typed String I
class Solution {
public:
    int possibleStringCount(string word) {
        int cnt=0,ans=1;
        for(int i=1;i<word.size();i++){
            if(word[i]==word[i-1]){cnt++;}
            else{
                ans+=cnt;
                cnt=0;
            }
        }
        ans+=(cnt);
        return ans;
    }
};
