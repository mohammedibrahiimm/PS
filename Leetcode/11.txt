// https://leetcode.com/problems/container-with-most-water/
// 11. Container With Most Water
class Solution {
public:
    int maxArea(vector<int>& h) {
        int l=0,r=h.size()-1;
        int ans=0;
        while(l<r){
            ans=max(ans,min(h[r],h[l])*(r-l));
            if(h[r]>=h[l]){l++;}
            else{r--;}
        }
        return ans;
    }
};
