// 118. Pascal's Triangle
class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>>v(numRows);
        v[0]={1};
        for(int i=1;i<numRows;i++){
            v[i]=vector<int>(v[i-1].size()+1,1);
            for(int j=1;j<v[i].size()-1;j++){
                v[i][j]=v[i-1][j-1]+v[i-1][j];
            }
        }
        return v;
    }
};
