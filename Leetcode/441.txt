// https://leetcode.com/problems/arranging-coins/
// 441. Arranging Coins
class Solution {
public:
    int arrangeCoins(int n) {
        long long l=0,r=65535;
        while(l<=r){
            long long m=(l+r)>>1;
            if((m*(m+1)>>1)<=n){l=m+1;}
            else{r=m-1;}
        } 
        return (int)r;
    }
};
