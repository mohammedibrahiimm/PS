-- 1304. Find N Unique Integers Sum up to Zero
class Solution {
public:
    vector<int> sumZero(int n) {
        int num=1000;
        vector<int>ans;
        while(n>1){
            ans.push_back(num);
            ans.push_back(-num);
            n-=2;
            num--;
        }
        if(n){ans.push_back(0);}
        return ans;
    }
};
