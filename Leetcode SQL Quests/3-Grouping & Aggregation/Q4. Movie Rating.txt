-- Q4. Movie Rating
(select name as results from users u
join MovieRating mr
on mr.user_id = u.user_id 
group by name
order by count(name) desc, name asc
limit 1)
union all
(select title as results from movies m
join MovieRating mr
on m.movie_id = mr.movie_id and month(created_at)=2 and year(created_at)=2020  
group by title 
order by avg(rating) desc, title asc
limit 1)
