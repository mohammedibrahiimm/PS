-- Q1. Trips and Users
select request_at as "Day",round(sum(case when status="completed" then 0 else 1 end)/count(*),2) as "Cancellation Rate"
from Trips t join Users u1
on t.client_id=u1.users_id and u1.banned!="Yes"
join Users u2 on t.driver_id=u2.users_id and u2.banned!="Yes"
where request_at between "2013-10-01" and "2013-10-03"
group by request_at
order by request_at
