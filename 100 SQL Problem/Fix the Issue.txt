-- Fix the Issue
SELECT LOWER(TRIM(product_name)) as product_name, TO_CHAR(sale_date, 'YYYY-MM') as sale_date,count(*) as total
FROM sales
group by LOWER(TRIM(product_name)) , TO_CHAR(sale_date, 'YYYY-MM')
order by product_name asc,sale_date asc
