-- All Valid Triplets That Can Represent a Country
select SchoolA.student_name as member_A,SchoolB.student_name as member_B,SchoolC.student_name as member_c from SchoolA 
join SchoolB 
on SchoolA.student_id != SchoolB.student_id and SchoolA.student_name != SchoolB.student_name
join SchoolC
on SchoolA.student_id != Schoolc.student_id and SchoolA.student_name != Schoolc.student_name
and Schoolc.student_id != SchoolB.student_id and Schoolc.student_name != SchoolB.student_name
