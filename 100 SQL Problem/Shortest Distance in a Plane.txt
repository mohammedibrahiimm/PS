-- Shortest Distance in a Plane

select distinct abs(t1.x-t2.x)+abs(t1.y-t2.y) as shortest from point_2d t1
join point_2d t2
on abs(t1.x-t2.x)+abs(t1.y-t2.y)!=0
order by shortest
limit 1
