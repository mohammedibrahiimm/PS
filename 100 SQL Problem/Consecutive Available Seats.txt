-- Consecutive Available Seats
with tmp as(
    SELECT seat_id,free,
    ROW_NUMBER() OVER (ORDER BY seat_id) AS row_number
    FROM cinema 
)
select t1.seat_id from tmp t1
join tmp t2 on t1.row_number+1=t2.row_number and t1.free='t' and t2.free='t'
