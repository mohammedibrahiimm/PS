-- New Users Daily Count
with tmp as (select user_id,min(activity_date) from Traffic
where activity='login'
group by user_id  
having '2019-06-30'-min(activity_date)<=90 and '2019-06-30'-min(activity_date)>=0
)

select min as login_date,count(min) as user_count
from tmp
group by login_date
order by login_date
