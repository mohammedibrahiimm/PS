-- Maximum Transaction Each Day
with new as(
    select date(day) as vis_date,max(amount) as max_amount from Transactions
    group by date(day)
)
select transaction_id from Transactions t1
join new n1
on date(t1.day)=n1.vis_date and t1.amount=n1.max_amount
order by transaction_id
