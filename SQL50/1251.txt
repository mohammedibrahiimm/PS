-- 1251. Average Selling Price
select p.product_id,ifnull(round(sum(p.price*u.units)/sum(u.units),2),0) as average_price from prices p
left join UnitsSold u on p.product_id=u.product_id and p.start_date <= u.purchase_date and u.purchase_date <= p.end_date
group by product_id
